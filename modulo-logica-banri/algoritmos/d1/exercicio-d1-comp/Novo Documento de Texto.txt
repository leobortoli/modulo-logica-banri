using System;

namespace complementar
{
    class Metrica
    {
        class Terreno
        {
            public double comprimento;
            public double largura;
            
            public double GetArea()
            {
                return this.comprimento * this.largura;
            }
        }

        class Casa
        {
            public double comprimento;
            public double largura;
            public double GetArea()
            {
                return this.comprimento * this.largura;
            }
        }

        static void Main(string[] args)
        {
            Terreno terra = new Terreno();

            Casa lar = new Casa();

            terra.comprimento = 500;
            terra.largura = 200;

            lar.comprimento = 200;
            lar.largura = 100;

            double areaLivre = terra.GetArea() - lar.GetArea();

            Console.WriteLine($"Terra: \nComprimento: {terra.comprimento} \nLargura: {terra.largura} \n");
            Console.WriteLine($"Propriedade: \nComprimento: { lar.comprimento} \nLargura: { lar.largura}");
            Console.WriteLine($"Area Livre: {areaLivre}");
        }
    }
}

using System;

namespace complementar
{
    class Temporizador
    {
        static void Main(string[] args)
        {
            int segundoPassado = Convert.ToInt32(Console.ReadLine());
            int minutoPassado = Convert.ToInt32(Console.ReadLine());
            int HoraPassada = Convert.ToInt32(Console.ReadLine());
            int segundo = 1;
            int minuto = 60;
            int hora = 3600;

            int tempoTotal = 24*hora;

            int tempoPassado = (segundoPassado * segundo) + (minutoPassado * minuto) + (HoraPassada * hora);

            int tempoRestante = tempoTotal - tempoPassado;

            int horaRestante = tempoRestante/hora;

            int tempoAtualizado = tempoRestante - (horaRestante*hora);

            int minutoRestante = tempoAtualizado / minuto;

            int segundoRestante = tempoAtualizado - (minutoRestante*minuto);

            Console.WriteLine($"Tempo Passado (em segundos): {tempoPassado}");
            Console.WriteLine($"Tempo Restante (em segundos): {tempoRestante}");
            Console.WriteLine($"Tempo Restante: {horaRestante}:{minutoRestante}:{segundoRestante}");
        }
    }
}
