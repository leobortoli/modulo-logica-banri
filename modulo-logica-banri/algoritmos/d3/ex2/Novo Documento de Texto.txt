using System;
using System.Globalization;

class AssistenteVirtual
{
    static DateTime dataLocal = DateTime.Now;
    static readonly CultureInfo cultura = new CultureInfo("pt-BR");

    static void Main()
    {
        int opcaoSelecionada;
        do 
        {
            ExibirMenu();
            opcaoSelecionada = int.Parse(Console.ReadLine());
            switch (opcaoSelecionada)
            {
                case 0:
                    break;
                case 1:
                    ObterDataAtual(dataLocal, cultura);
                    break;
                case 2:
                    ObterHoraAtual(dataLocal, cultura);
                    break;
                case 3:
                    DizerOla();
                    break;
                default:
                    break;
            }
        } while (opcaoSelecionada != 0);
    }

    {
        Console.WriteLine("===== Menu Interativo =====");
